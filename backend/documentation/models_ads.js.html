"use strict";
const { Model } = require("sequelize");
module.exports = (sequelize, DataTypes) => {
  class ads extends Model {
    /**
     * Helper method for defining associations.
     * This method is not a part of Sequelize lifecycle.
     * The `models/index` file will call this method automatically.
     */

    static associate(models) {
      ads.belongsTo(models.User, {
        foreignKey: "userId",
        onDelete: "CASCADE",
      });
      ads.hasMany(models.favourites, {
        foreignKey: "adId",
      });
      ads.hasMany(models.appointments, {
        foreignKey: "adId",
      });
      // define association here
    }
  }
  ads.init(
    {
      title: {

        allowNull: false,

        allowNull: true,

      audience: {
        type: DataTypes.ENUM,
        values: ["University", "School"],
      },
      university: {
        type: DataTypes.STRING,
        allowNull: false,
      },
      major: {
        type: DataTypes.STRING,
        allowNull: false,
      },
      class: {
        type: DataTypes.STRING,
        allowNull: false,
      },
      hashtags: {
        type: DataTypes.STRING,
        allowNull: false,
      },
      description: {
        type: DataTypes.TEXT,
        allowNull: false,
      },
      hourlyRate: {
        type: DataTypes.STRING,
        allowNull: false,

        type: DataTypes.STRING,
        allowNull: false,
      },
      studyNotes: {
        type: DataTypes.ENUM,
        values: ["Yes", "No"],
        defaultValue: "No",

        type: DataTypes.STRING,
